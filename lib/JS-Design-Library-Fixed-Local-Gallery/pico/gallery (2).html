Dialog מתוקן

החלון נפתח רק בלחיצה ונסגר בלחיצה על אישור או ביטול.

אישור פעולה

האם להמשיך?

 
<button data-open-dialog="#myDialog">פתיחת חלון</button>

<dialog id="myDialog">
  <article>
    <h3>אישור פעולה</h3>
    <p>האם להמשיך?</p>
    <footer>
      <button type="button" data-close-dialog>ביטול</button>
      <button type="button" data-close-dialog>אישור</button>
    </footer>
  </article>
</dialog>

<script>
document.addEventListener("click", (event) => {
  const openButton = event.target.closest("[data-open-dialog]");
  if (openButton) {
    document.querySelector(openButton.dataset.openDialog).showModal();
  }

  const closeButton = event.target.closest("[data-close-dialog]");
  if (closeButton) {
    closeButton.closest("dialog").close();
  }
});
</script>
